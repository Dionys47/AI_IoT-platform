void setup() {
  // Lidh WiFi-në dhe inicializon DHT22
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  dht.begin();
}

void loop() {
  // Lexon sensorin çdo 10 sekonda
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  sendData(t, "temperature", "C");
  sendData(h, "humidity", "%");
}

void sendData(float value, const char* type, const char* unit) {
  // POST në /api/v1/ingest me headers X-Device-Id dhe X-Device-Key
  http.addHeader("X-Device-Id", DEVICE_ID);

  http.POST("{ \"value\": 23.5, \"sensor_type\": \"temperature\", \"unit\": \"C\" }");
}
